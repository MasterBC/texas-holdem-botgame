26       * Creates a map where cards are organized according to their Rank.
27       * 
28       * @param cards
29       * @return a map where cards are organized according to their Rank
30       */
31      protected static Map<Rank, List<Card>> getRankDistribution(
32              final List<Card> cards) {
33  
34          final Map<Rank, List<Card>> distribution = new HashMap<Rank, List<Card>>();
35          for (final Card c : cards) {
36              if (distribution.containsKey(c.rank())) {
37                  distribution.get(c.rank()).add(c);
38              } else {
39                  final ArrayList<Card> l = new ArrayList<Card>();
40                  l.add(c);
41                  distribution.put(c.rank(), l);
42              }
43          }
44          return distribution;
45      }

92      /**
93       * Creates a new list containing only the cards in target that don't exist
94       * in remove. Null safe. If target is empty or null an empty list will be
95       * returned.
96       * 
97       * @param target
98       * @param remove
99       * @return A copy of the target list with all elements from remove removed.
100      */
101     protected static List<Card> remove(
102             final List<Card> target,
103             final List<Card> remove) {
104 
105         final List<Card> result = new ArrayList<Card>();
106 
107         if (CollectionUtils.isEmpty(remove)) {
108             result.addAll(target);
109             return result;
110         }
111 
112         for (final Card c : target) {
113             if (!remove.contains(c)) {
114                 result.add(c);
115             }
116         }
117 
118         return result;
119     }

121     /**
122      * Sorts the list of cards by Rank in descending order and removes any
123      * occurrences of cards in the list of excluded cards.
124      * 
125      * Example: getHighestSortedAndExclude(2, {5d, 9c, Jd, 2c, Qs}, {Qs, 9c})
126      * Result: {Jd, 5d}
127      * 
128      * @param noof
129      *            The max length of the resulting list of cards
130      * @param cards
131      * @param exclude
132      *            A list of cards to exclude from the result
133      * @return A descending (by Rank) list of cards
134      */
135     protected static List<Card> getHighestSortedAndExclude(
136             final int noof,
137             final List<Card> cards,
138             final List<Card> exclude) {
139 
140         final List<Card> removed = remove(cards, exclude);
141         Collections.sort(removed, CardSortBy.RANK.getComparator());
142         Collections.reverse(removed);
143 
144         if (noof > removed.size())
145             return removed;
146 
147         return removed.subList(0, noof);
148     }

166     /**
167      * Returns the longest possible list of cards in Rank order. A card ranked
168      * ACE can be both at the beginning of a list or at the end. If no
169      * consecutive list can be found in the list of cards an empty list is
170      * returned.
171      * 
172      * Example: getLongestConsecutiveSubset({5h, 3s, Ad, As, 2c, Kh, 4c})
173      * Returns: {Ad, 2c, 3s, 4c, 5h}
174      * 
175      * @param cards
176      * @return The longest consecutive sublist of cards found or the empty list
177      *         if none found.
178      */
179     protected static List<Card> getLongestConsecutiveSubset(
180             final List<Card> cards) {
181 
182         final List<Card> sortedCards = removeDuplicatesByRankAndSortByRank(cards);
183 
184         // If an ACE exists in cards place it at beginning as well since
185         // in a straight it may act as numeric value 14 or 1.
186         final Card lastCard = sortedCards.get(sortedCards.size() - 1);
187         if (lastCard.rank() == Rank.ACE) {
188             sortedCards.add(0, lastCard);
189         }
190 
191         List<Card> largestConsecutive = new ArrayList<Card>();
192 
193         int currStartPos = 0;
194         Card previousCard = null;
195         for (int i = 0; i < sortedCards.size(); i++) {
196             final Card currentCard = sortedCards.get(i);
197 
198             if (previousCard == null ||
199                     // If the previous card value + 1 is not equal to the
200                     // current card's value the order is broken. Unless...
201                     ((previousCard.rank().getOrderValue() + 1 != currentCard
202                             .rank().getOrderValue()) &&
203                     // ... the previous card was an ACE and the current
204                     // card is valued 2.
205                     !(previousCard.rank() == Rank.ACE && currentCard.rank()
206                             .getOrderValue() == 2))) {
207 
208                 // Found consecutive subset larger than one,
209                 // check if it is longer than the previously
210                 // known one.
211                 if (currStartPos < i - 1
212                         && i - currStartPos > largestConsecutive.size() - 1)
213                     largestConsecutive = sortedCards.subList(currStartPos, i);
214 
215                 currStartPos = i;
216             }
217 
218             previousCard = currentCard;
219         }
220 
221         // In case the end is part of the largest consecutive sublist
222         if (currStartPos < sortedCards.size() - 1
223                 && sortedCards.size() - 1 - currStartPos > largestConsecutive
224                         .size() - 1)
225             largestConsecutive = sortedCards.subList(currStartPos,
226                     sortedCards.size());
227 
228         return largestConsecutive;
229     }

231     /**
232      * Sorts the list of cards by Rank and then removes any cards of the same
233      * Rank.
234      * 
235      * Example: removeDuplicatesByRankAndSortByRank({5h, Jd, 3s, 5d, Js}
236      * Returns: {3s, 5d, Jd}
237      * 
238      * @param cards
239      * @return A sorted list of Cards with no duplicate Ranks.
240      */
241     protected static List<Card> removeDuplicatesByRankAndSortByRank(
242             final List<Card> cards) {
243 
244         final List<Card> sortedCards = sortBy(CardSortBy.RANK, cards);
245         final List<Card> result = new ArrayList<Card>();
246         Card previousCard = null;
247 
248         for (final Card c : sortedCards) {
249             if (previousCard == null || previousCard.rank() != c.rank())
250                 result.add(c);
251 
252             previousCard = c;
253         }
254         return result;
255     }

286     /**
287      * Searches a List of Cards for cards of the same rank that exist in least
288      * noof times and returns the highest ranking list.
289      * 
290      * Example: getHighestOfSameRank(2, {5h, Jd, 3s, 5d, Js} Returns: {Jd, Js}
291      * 
292      * @param noof
293      *            The minimum number of cards of same rank
294      * @param cards
295      *            The list of cards available
296      * @return A list of cards of noof length of the highest rank.
297      */
298     protected static List<Card> getHighestOfSameRank(
299             final int noof,
300             final List<Card> cards) {
301 
302         List<Card> result = new ArrayList<Card>();
303 
304         Rank currBestRank = null;
305         final Map<Rank, List<Card>> distribution = getRankDistribution(cards);
306         for (final Entry<Rank, List<Card>> entry : distribution.entrySet()) {
307             if (entry.getValue().size() >= noof) {
308                 if (currBestRank == null
309                         || currBestRank.compareTo(entry.getKey()) < 0) {
310                     currBestRank = entry.getKey();
311                     result = entry.getValue();
312                 }
313             }
314         }
315         Collections.sort(result, CardSortBy.SUIT.getComparator());
316         if (result.size() > noof)
317             result = result.subList(0, noof);
318 
319         return result;
320     }
