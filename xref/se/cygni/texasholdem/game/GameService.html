358     public void connect() throws IOException {
359 
360         final RpcServerCallExecutor executor = new ThreadPoolCallExecutor(3, 10);
361 
362         bootstrap = new DuplexTcpClientBootstrap(
363                 clientPeerInfo,
364                 new NioClientSocketChannelFactory(
365                         Executors.newCachedThreadPool(),
366                         Executors.newCachedThreadPool()),
367                 executor);
368 
369         // Configure the client
370         final Service clientService = ClientServer.PlayerService
371                 .newReflectiveService(playerService);
372         bootstrap.getRpcServiceRegistry().registerService(clientService);
373 
374         // Set up the event pipeline factory.
375         bootstrap.setOption("connectTimeoutMillis", 10000);
376         bootstrap.setOption("connectResponseTimeoutMillis", 10000);
377         bootstrap.setOption("sendBufferSize", 1048576);
378         bootstrap.setOption("receiveBufferSize", 1048576);
379         bootstrap.setOption("tcpNoDelay", false);
380 
381         final CleanShutdownHandler shutdownHandler = new CleanShutdownHandler();
382         shutdownHandler.addResource(bootstrap);
383 
384         // setup a RPC event listener - it just logs what happens
385         final RpcConnectionEventNotifier rpcEventNotifier = new RpcConnectionEventNotifier();
386         final RpcConnectionEventListener listener = new RpcConnectionEventListener() {
387 
388             @Override
389             public void connectionReestablished(
390                     final RpcClientChannel clientChannel) {
391 
392                 log.info("connectionReestablished " + clientChannel);
393                 player.connectionToGameServerEstablished();
394             }
395 
396             @Override
397             public void connectionOpened(final RpcClientChannel clientChannel) {
398 
399                 log.info("connectionOpened " + clientChannel);
400                 player.connectionToGameServerEstablished();
401             }
402 
403             @Override
404             public void connectionLost(final RpcClientChannel clientChannel) {
405 
406                 log.info("connectionLost " + clientChannel);
407                 player.connectionToGameServerLost();
408             }
409 
410             @Override
411             public void connectionChanged(final RpcClientChannel clientChannel) {
412 
413                 log.info("connectionChanged " + clientChannel);
414             }
415         };
416         rpcEventNotifier.setEventListener(listener);
417         bootstrap.registerConnectionEventListener(rpcEventNotifier);
418 
419         // Connect to server
420         serverChannel = bootstrap.peerWith(serverPeerInfo);
421         gameService = se.cygni.texasholdem.communication.ClientServer.GameService
422                 .newBlockingStub(serverChannel);
423     }
