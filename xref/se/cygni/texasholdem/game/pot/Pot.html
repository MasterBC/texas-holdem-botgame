37      /**
38       * Calculates the bet the player has to make to Call. If player has folded 0
39       * is returned.
40       * 
41       * @param player
42       * @return
43       */
44      public long getMinimumBetForPlayerToCall(final BotPlayer player) {
45  
46          if (hasFolded(player))
47              return 0L;
48  
49          final long max = getMaxTotalBetInCurrentPlayState();
50          final long playerBet = getTotalBetAmountForPlayerInPlayState(player,
51                  currentPlayState);
52          return max - playerBet;
53      }

55      /**
56       * Places a bet for the player.
57       * 
58       * @throws IllegalStateException
59       *             if in a state that doesn't allow bets
60       * @throws IllegalStateException
61       *             if player has folded
62       * @throws IllegalArgumentException
63       *             if bet is negative
64       * @param player
65       * @param amount
66       */
67      public void bet(final BotPlayer player, final long amount) {
68  
69          if (!canPlaceBetInCurrentPlayState())
70              throw new IllegalStateException("Pot is in state: "
71                      + currentPlayState.getName() + ", no more bets allowed");
72  
73          if (foldedPlayers.contains(player))
74              throw new IllegalStateException("Player: " + player
75                      + " has folded, cannot bet");
76  
77          if (amount < 0)
78              throw new IllegalArgumentException("Player: " + player
79                      + " tried to place negative bet");
80  
81          if (amount == 0)
82              return;
83  
84          if (!transactionTable.containsKey(currentPlayState)) {
85              transactionTable.put(currentPlayState,
86                      new ArrayList<PotTransaction>());
87          }
88  
89          final PotTransaction transaction = new PotTransaction(
90                  transactionCounter.getAndIncrement(), player, amount);
91          transactionTable.get(currentPlayState).add(transaction);
92      }

138     /**
139      * Calculates the total amount in the Pot.
140      * 
141      * @return
142      */
143     public long getTotalPotAmount() {
144 
145         long total = 0;
146 
147         for (final Entry<PlayState, List<PotTransaction>> entry : transactionTable
148                 .entrySet()) {
149             for (final PotTransaction transaction : entry.getValue()) {
150                 total += transaction.getAmount();
151             }
152         }
153 
154         return total;
155     }

157     /**
158      * Calculates the total bet amount per player in the current PlayState and
159      * returns the highest.
160      * 
161      * @return
162      */
163     public long getMaxTotalBetInCurrentPlayState() {
164 
165         long highest = 0;
166 
167         for (final BotPlayer player : allPlayers) {
168             if (foldedPlayers.contains(player))
169                 continue;
170 
171             final long playerTotal = getTotalBetAmountForPlayerInPlayState(
172                     player, currentPlayState);
173             highest = highest < playerTotal ? playerTotal : highest;
174         }
175 
176         return highest;
177     }

179     /**
180      * Calculates the total amount for all bets placed by Player during this
181      * whole play.
182      * 
183      * @param player
184      * @return
185      */
186     public long getTotalBetAmountForPlayer(final BotPlayer player) {
187 
188         long total = 0;
189         PlayState state = currentPlayState;
190         total += getTotalBetAmountForPlayerInPlayState(player, state);
191 
192         while (PlayState.hasPreviousState(state)) {
193             state = PlayState.getPreviousState(state);
194             total += getTotalBetAmountForPlayerInPlayState(player, state);
195         }
196 
197         return total;
198     }

200     /**
201      * Calculates the amount of all the bets placed by the Player in specified
202      * PlayState.
203      * 
204      * @param player
205      * @param playState
206      * @return
207      */
208     public long getTotalBetAmountForPlayerInPlayState(
209             final BotPlayer player,
210             final PlayState playState) {
211 
212         final List<PotTransaction> transactions = transactionTable
213                 .get(playState);
214         if (CollectionUtils.isEmpty(transactions))
215             return 0L;
216 
217         long total = 0;
218         for (final PotTransaction transaction : transactions) {
219             if (transaction.getPlayer().equals(player))
220                 total += transaction.getAmount();
221         }
222 
223         return total;
224     }

226     /**
227      * Checks that all active players have placed bets amounting to the same
228      * value in the current play state.
229      * 
230      * @return
231      */
232     public boolean isCurrentPlayStateBalanced() {
233 
234         Long lastTotal = null;
235         for (final BotPlayer player : allPlayers) {
236             if (foldedPlayers.contains(player))
237                 continue;
238 
239             final long playerTotal = getTotalBetAmountForPlayerInPlayState(
240                     player, currentPlayState);
241             if (lastTotal == null)
242                 lastTotal = Long.valueOf(playerTotal);
243             else if (lastTotal != playerTotal)
244                 return false;
245         }
246 
247         return true;
248     }

255     protected boolean canPlaceBetInCurrentPlayState() {
256 
257         return PlayState.hasNextState(currentPlayState);
