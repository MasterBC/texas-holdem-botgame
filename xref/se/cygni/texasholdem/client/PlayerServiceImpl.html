192     @Override
193     public void onYouMustAct(
194             final RpcController controller,
195             final PBPossibleActions request,
196             final RpcCallback<PBAction> done) {
197 
198         try {
199             final List<Action> possibleActions = new ArrayList<Action>();
200 
201             for (final PBAction pbAction : request.getActionsList()) {
202                 possibleActions.add(ConversionUtil.convertPBAction(pbAction));
203             }
204 
205             final Action action = player.onActionRequired(possibleActions);
206 
207             log.debug("Returning action {} amount {}", action.getActionType(),
208                     action.getAmount());
209 
210             done.run(ConversionUtil.convertAction(action));
211 
212         } catch (final Exception e) {
213             log.error("Failed to reply with an action", e);
214             controller.setFailed(e.getMessage());
215         }
216     }
