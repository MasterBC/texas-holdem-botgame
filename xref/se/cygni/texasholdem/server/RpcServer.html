38              final GameServiceImpl gameService, final GameServer gameServer) {
39  
40          this.serverHostname = serverHostname;
41          this.serverPort = serverPort;
42          this.gameService = gameService;
43          this.gameServer = gameServer;

46      private void initServer() {
47  
48          final PeerInfo serverInfo = new PeerInfo(serverHostname, serverPort);
49  
50          final RpcServerCallExecutor executor = new ThreadPoolCallExecutor(3, 10);
51  
52          // Configure the server.
53          bootstrap = new DuplexTcpServerBootstrap(
54                  serverInfo,
55                  new NioServerSocketChannelFactory(
56                          Executors.newCachedThreadPool(),
57                          Executors.newCachedThreadPool()),
58                  executor);
59  
60          bootstrap.setOption("sendBufferSize", 1048576);
61          bootstrap.setOption("receiveBufferSize", 1048576);
62          bootstrap.setOption("child.receiveBufferSize", 1048576);
63          bootstrap.setOption("child.sendBufferSize", 1048576);
64          bootstrap.setOption("tcpNoDelay", false);
65  
66          final CleanShutdownHandler shutdownHandler = new CleanShutdownHandler();
67          shutdownHandler.addResource(bootstrap);
68  
69          final RpcConnectionEventNotifier rpcEventNotifier = new RpcConnectionEventNotifier();
70          rpcEventNotifier.setEventListener(createRpcConnectionEventListener());
71          bootstrap.registerConnectionEventListener(rpcEventNotifier);
72  
73          // Register the GameService
74          final Service gamePbService = ClientServer.GameService
75                  .newReflectiveService(gameService);
76          bootstrap.getRpcServiceRegistry().registerService(gamePbService);
77      }
78  
79      @PostConstruct
80      public void startServer() {
81  
82          initServer();
83          bootstrap.bind();
84  
85          log.debug("RpcServer started for host name: {}, listening to port: {}",
